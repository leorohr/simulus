10     
11     import java.util.ArrayList;
12     
13     public class Tile extends Group {
14     
15     	private int gridPosX;
16     	private int gridPosY;
17     	private boolean isOccupied;
18     	private Vehicle occupier;
19     	private ArrayList<Path> turningPaths;
20         protected Rectangle frame;
21         protected boolean isRedLight = false;
22     
23     	public Tile(double posX, double posY, double width, double height,
24     			int gridPosX, int gridPosY) {
25     		//super(posX, posY, width, height);
26     		frame = new Rectangle(posX, posY, width, height);
27     		this.gridPosX = gridPosX;
28     		this.gridPosY = gridPosY;
29     		occupier = null;
30     		isOccupied = false;
31     		turningPaths = new ArrayList<>();
32     		frame.setFill(new ImagePattern(ResourceBuilder.getLandTexture()));
33     //		frame.setStroke(Color.BLACK);
34     //		frame.setStrokeWidth(0.2d);
35     		this.getChildren().add(frame);
36     	}
37     
38         //Makes sure that only the current occupier can set the tile's occupation status
39     	public void setOccupied(boolean isOccupied, Vehicle occupier) {
40     		if (isOccupied) {
41     			this.occupier = occupier;
42     			this.isOccupied = true;
43     			redrawTile();
44     		}else if(occupier.equals(getOccupier())){
45     			this.occupier = null;
46     			this.isOccupied = false;
47     			redrawTile();
48     		}
49     	}
50     	
51     	public void setOccupied(boolean isOccupied){
52     		this.isOccupied = isOccupied;
53     		if(isOccupied)
54     			isRedLight = true;
55     		else isRedLight = false;
56     		redrawTile();
57     	}
58     
59     	public Vehicle getOccupier() {
60     		return occupier;
61     	}
62     
63     	public boolean isOccupied() {
64     		return isOccupied;
65     	}
66     
67     	public int getGridPosX() {
68     		return gridPosX;
69     	}
70     
71     	public int getGridPosY() {
72     		return gridPosY;
73     	}
74     	
75     	public boolean getRedLight(){
76     		return isRedLight;
77     	}
78     
79     	/**
80     	 * If the app is currently in debug mode, this method draws
81     	 * the tiles occupied by vehicles in green.
82     	 */
83     	public void redrawTile() {
84             if(!SimulationController.getInstance().isDebug())
85     			return;
86     		
87     		if (isOccupied()) {
88                 frame.setFill(Color.GREEN);
89                 if(isRedLight)
90                 	frame.setFill(Color.RED);
91     		} else {
92                 if(this instanceof Lane)
93     			    frame.setFill(Lane.COLOR);
94                 else frame.setFill(Color.TRANSPARENT);
95                 if(isRedLight)
96                 	frame.setFill(Color.RED);
97     		}
98     	}
99     	
100     	/**
101     	 * @return The x-coordinate of the top left corner of the tile.
102     	 */
103     	public double getX(){
104     		return frame.getX();
105     	}
106     	
107     	/**
108     	 * @return The y-coordinate of the top left corner of the tile.
109     	 */
110     	public double getY(){
111     		return frame.getY();
112     	}
113     	
114     	public double getWidth(){
115     		return frame.getWidth();
116     	}
117     	
118     	public double getHeight(){
119     		return frame.getHeight();
120     	}
121     	
122     	public Rectangle getFrame(){
123     		return frame;
124     	}
125     
126     	public double getCenterX() {
127     		return getX() + (getWidth() / 2);
128     	}
129     
130     	public double getCenterY() {
131     		return getY() + (getHeight() / 2);
132     	}
133     	
134     	/**
135     	 * @return The list of available paths for cars to take when they face this tile in an intersection.
136     	 */
137     	public ArrayList<Path> getTurningPaths() {
138     		return turningPaths;
139     				
140     	}
141     	public String toString(){
142     		return "X: " + gridPosX + " Y: " + gridPosY;
143     	}
