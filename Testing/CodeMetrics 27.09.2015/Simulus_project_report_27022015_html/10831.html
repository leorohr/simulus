105         void addNumCarData(int num) {
106     
107             numCarsSeries.getData().add(new LineChart.Data<>(dataCount++, num));
108             ((NumberAxis)numCarsChart.getXAxis()).setLowerBound((0 > dataCount-MAX_DATA_POINTS ? 0 : dataCount-MAX_DATA_POINTS));
109             ((NumberAxis)numCarsChart.getXAxis()).setUpperBound(dataCount);
110     
111             if(numCarsSeries.getData().size() > MAX_DATA_POINTS) {
112                 numCarsSeries.getData().remove(0);
113             }
