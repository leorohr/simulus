1     package com.simulus.view;
2     
3     import com.simulus.util.enums.Direction;
4     import com.simulus.util.enums.Seed;
5     import javafx.scene.Group;
6     
7     import java.util.Arrays;
8     import java.util.List;
9     
10     /**
11      * Represents a roadblock. Roadblock refers to four adjacent lanes, hence one roadblock takes up a 1x4 space
12      * in the grid of tiles.
13      */
14     public class Road extends Group implements TileGroup{
15     
16     	private Lane[] lanes = new Lane[4];
17     	private Seed orientation;
18     	
19     	/**
20     	 * @param gridPosX The mostleft/top x-coordinate of the road in the grid.
21     	 * @param gridPosY The mostleft/top y-coordinate of the road in the grid.
22     	 * @param orientation The orientation of the Road.
23     	 */
24     	public Road(int gridPosX, int gridPosY, Seed orientation){
25     		
26     		int tileSize = Map.TILESIZE;
27     		this.orientation = orientation;
28     		
29     		switch(orientation){
30     		case NORTHSOUTH:
31     			for (int i = 0; i < lanes.length; i++) {
32     				if(i<2) 
33     					lanes[i] = new Lane((gridPosX + i)*tileSize, gridPosY*tileSize,
34     										tileSize, tileSize, gridPosX+i, gridPosY, Direction.NORTH, i);
35     				else
36     					lanes[i] = new Lane((gridPosX + i)*tileSize, gridPosY * tileSize,
37     										tileSize, tileSize, gridPosX+i, gridPosY, Direction.SOUTH, i);
38     			}
39     			break;
40     		case WESTEAST:
41     			for (int i = 0; i < lanes.length; i++) {
42     				if(i<2) 
43     					lanes[i] = new Lane(gridPosX*tileSize, (gridPosY + i)*tileSize,
44     										tileSize, tileSize, gridPosX, gridPosY+i, Direction.EAST, i);
45     				else
46     					lanes[i] = new Lane(gridPosX*tileSize, (gridPosY + i)*tileSize,
47     										tileSize, tileSize, gridPosX, gridPosY+i, Direction.WEST, i);
48     			}
49     			break;
50     		default:
51     			break;
52     		}
53     		
54     		this.getChildren().addAll(lanes);
55     	}
56     
57         /**
58          * @return A list containing the 4 tiles that this roadblock is comprised of.
59          */
60     	@Override
61     	public List<Tile> getTiles() {
62     		
63     		return Arrays.asList(lanes);
64     	}
65     
66         /**
67          * @return The orientation of this roadblock.
68          */
69     	public Seed getOrientation() {
70     		return this.orientation;
71     	}
72     }
