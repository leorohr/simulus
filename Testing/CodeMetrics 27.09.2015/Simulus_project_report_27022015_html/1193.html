22     package com.simulus.util.enums;
23     
24     public enum Seed {
25     	
26     	EMPTY, NORTHSOUTH, WESTEAST, INTERSECTION;
27     	
28     	/**
29     	 * @return Randomly returns one of the two direction included by the orientation or <code>null</code>
30     	 * if it is called on an intersection.
31     	 */
32     	public Direction randomDirection() {
33     		if(this == NORTHSOUTH) {
34     			if(Math.random() > 0.5)
35     				return Direction.NORTH;
36     			else return Direction.SOUTH;
37     		} else if(this == WESTEAST) {
38     			if(Math.random() > 0.5)
39     				return Direction.EAST;
40     			else return Direction.WEST;
41     		}
42     		
43     		return null;
44     	}

46     }
